PS.: You need to import the KoalaDialogModule in your component module.

page-request.component.ts code 
import { Component } from "@angular/core";
import { KoalaRequestService } from "@koalarx/ui/core";
import { Observable } from "rxjs";

interface ResponseInterface {
  message: string;
}

@Component({
  templateUrl: 'page-request.component.html'
})
export class PageRequestComponent {

  constructor(private requestService: KoalaRequestService) {}

  public requestSuccess() {
    this.requestService
        .request<ResponseInterface>(
          new Observable<ResponseInterface>(observe => observe.next({
            message: 'This request run with success!'
          })),
          response => alert(response.message),
          error => alert(error.message)
        );
  }

  public requestError() {
    this.requestService
        .request<ResponseInterface>(
          new Observable<ResponseInterface>(observe => observe.error({
            message: 'Error'
          })),
          response => alert(response.message),
          error => alert(error.message)
        );
  }
}
