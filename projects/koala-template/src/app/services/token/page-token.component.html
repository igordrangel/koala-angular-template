page-token.component.ts code 
import { Component } from "@angular/core";
import { KoalaTokenService } from "ngx-koala";

@Component({
  templateUrl: 'page-token.component.html'
})
export class PageTokenComponent {
  constructor(private tokenService: KoalaTokenService) {}
}

setToken

This method set a Token on LocalStorage and redirect to defaultPage defined on app.component.html in koala-page.

this.tokenService.setToken(token);

getDecodedToken

Return a decoded token.

this.tokenService.getDecodedToken<UserInterface>();

getToken

Return a BehaviorSubject for when a token will be defined return a token on subscribe.

this.tokenService.getToken().subscribe(token => console.log(token));

removeToken

This method logout the user removing the token from LocalStorage.

this.tokenService.removeToken();
